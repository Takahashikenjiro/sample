Sub Delete1()
            
 Dim column As Range, brow As Range
 Dim maxrow As Long
 Dim j As Long
 Worksheets("統合").Activate
            
            
 maxrow = Range("C65536").End(xlUp).Row
 Set column = ThisWorkbook.Sheets(1).Range("C3", Cells(maxrow, 3))
            
 j = j + Elase1(column, maxrow)
            
 maxrow = Range("B65536").End(xlUp).Row
 Set column = ThisWorkbook.Sheets(1).Range("B3", Cells(maxrow, 2))
            
 j = j + Elase1(column, maxrow)
            
Open ThisWorkbook.Path & "\1を消すマクロ結果.txt" For Output As #1
Print #1, j & "件"
Close #1
            
End Sub
Function Elase1(ByRef column As Range, ByVal maxrow As Integer) As Integer
            
 Dim n As Long
 Dim i As Long
            
 For i = 1 To maxrow
  If column.Cells(i) = 1 Then
   column.Cells(i) = ""
   n = n + 1
  End If
 Next i
            
 Elase1 = n
            
End Function
